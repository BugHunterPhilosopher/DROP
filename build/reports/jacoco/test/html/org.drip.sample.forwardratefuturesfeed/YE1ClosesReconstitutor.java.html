/**
 * YE1ClosesReconstitutor Cleanses, Transforms, and Re-constitutes the Formatted YE1 Closes Feed.
 *
 * @author Lakshmi Krishnamurthy
 */

public class YE1ClosesReconstitutor {

	public static final void main (
		final String[] args)
		throws Exception
	{
		EnvManager.InitEnv ("");

		int iSpotDateIndex = 0;
		int iLastPriceIndex = 2;
		int iFuturesExpiryDateIndex = 8;
		String strFeedLocation = "C:\\DRIP\\CreditAnalytics\\Daemons\\Feeds\\FundingFutures\\YE1_ClosesFormatted.csv";

		FundingFuturesClosesReconstitutor.RegularizeCloses (
			strFeedLocation,
			iSpotDateIndex,
			iLastPriceIndex,
			iFuturesExpiryDateIndex
		);
	}
}
