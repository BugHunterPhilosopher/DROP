	private org.drip.analytics.date.JulianDate reconcileStartDate()
	{
		if (null != _dtInterestAccrualStart) return _dtInterestAccrualStart;

		if (null != _dtFirstCoupon) return _dtFirstCoupon;

		if (null != _dtIssue) return _dtIssue;

		return _dtAnnounce;
	}
