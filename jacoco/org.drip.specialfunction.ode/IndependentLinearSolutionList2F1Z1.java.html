public class IndependentLinearSolutionList2F1Z1
{

	/**
	 * Generate the 2F1 Instance of IndependentLinearSolutionList at z = 1
	 * 
	 * @param regularHypergeometricEstimator 2F1 Regular Hyper-geometric Estimator
	 * 
	 * @return The 2F1 Instance of IndependentLinearSolutionList at z = 1
	 */

	public static final org.drip.specialfunction.ode.IndependentLinearSolutionList Generate (
		final org.drip.specialfunction.definition.RegularHypergeometricEstimator
			regularHypergeometricEstimator)
	{
		if (null == regularHypergeometricEstimator)
		{
			return null;
		}

		final double a = regularHypergeometricEstimator.a();

		final double b = regularHypergeometricEstimator.b();

		final double c = regularHypergeometricEstimator.c();

		try
		{
			if (org.drip.numerical.common.NumberUtil.IsInteger (c - a - b))
			{
				return null;
			}
		}
		catch (java.lang.Exception e)
		{
			e.printStackTrace();

			return null;
		}

		org.drip.specialfunction.ode.IndependentLinearSolutionList independentLinearSolutionList2F1Z1 = new
			org.drip.specialfunction.ode.IndependentLinearSolutionList();

		try
		{
			if (!independentLinearSolutionList2F1Z1.add (
				regularHypergeometricEstimator.albinate (
					a,
					b,
					1. + a + b - c,
					null,
					new org.drip.function.definition.R1ToR1 (null)
					{
						@Override public double evaluate (
							final double z)
							throws java.lang.Exception
						{
							return 1. - z;
						}
					}
				)
			))
			{
				return null;
			}

			if (!independentLinearSolutionList2F1Z1.add (
				regularHypergeometricEstimator.albinate (
					c - a,
					c - b,
					1. + c - a,
					new org.drip.function.definition.R1ToR1 (null)
					{
						@Override public double evaluate (
							final double z)
							throws java.lang.Exception
						{
							return java.lang.Math.pow (
								1. - z,
								c - a - b
							);
						}
					},
					new org.drip.function.definition.R1ToR1 (null)
					{
						@Override public double evaluate (
							final double z)
							throws java.lang.Exception
						{
							return 1. - z;
						}
					}
				)
			))
			{
				return null;
			}
		}
		catch (java.lang.Exception e)
		{
			e.printStackTrace();
		}

		return independentLinearSolutionList2F1Z1;
	}
}
