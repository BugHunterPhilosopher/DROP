public class GaussIntegralEulerMascheroniEstimate
{

	public static final void main (
		final String[] argumentArray)
		throws Exception
	{
		EnvManager.InitEnv ("");

		int termCount = 1000000;
		double[] zArray =
		{
			0.5,
			1.0,
			1.5,
			2.0,
			2.5,
			3.0,
			3.5,
			4.0,
			4.5,
			5.0,
			5.5,
			6.0,
			6.5,
			7.0,
			7.5,
			8.0,
			8.5,
			9.0,
			9.5,
		};

		R1ToR1IntegrandEstimator gaussEulerMascheroniIntegral = IntegralEstimate.GaussEulerMascheroni();

		CumulativeSeriesEstimator abramowitzStegun2007 =
			CumulativeSeriesEstimator.AbramowitzStegun2007 (termCount);

		System.out.println ("\t|--------------------------------------||");

		System.out.println ("\t|   GAUSS INTEGRAL DIGAMMA ESTIMATE    ||");

		System.out.println ("\t|--------------------------------------||");

		System.out.println ("\t|    L -> R:                           ||");

		System.out.println ("\t|        - z                           ||");

		System.out.println ("\t|        - Abramowitz-Stegun (2007)    ||");

		System.out.println ("\t|        - Gauss Euler-Mascheroni      ||");

		System.out.println ("\t|--------------------------------------||");

		for (double z : zArray)
		{
			System.out.println (
				"\t|" + FormatUtil.FormatDouble (z, 1, 1, 1.) + " => " +
				FormatUtil.FormatDouble (
					abramowitzStegun2007.evaluate (z),
					1, 10, 1.
				) + " | " + FormatUtil.FormatDouble (
					gaussEulerMascheroniIntegral.evaluate (z),
					1, 10, 1.
				) + " ||"
			);
		}

		System.out.println ("\t|--------------------------------------||");

		EnvManager.TerminateEnv();
	}
}
