public class SphericalFirstOrderMinusThree
{

	private static final void Besselj (
		final SphericalBesselFirstKindEstimator sphericalBesselFirstKindEstimator,
		final double[] zArray)
		throws Exception
	{
		System.out.println ("\t|--------------------------------------------------------------------------||");

		System.out.println ("\t|          -3 ORDER SPHERICAL BESSEL FIRST KIND INTEGRAL ESTIMATE          ||");

		System.out.println ("\t|--------------------------------------------------------------------------||");

		System.out.println ("\t|        L -> R:                                                           ||");

		System.out.println ("\t|                - z                                                       ||");

		System.out.println ("\t|                - Alpha Bessel Estimate Row                               ||");

		System.out.println ("\t|--------------------------------------------------------------------------||");

		double alpha = -3.;

		for (double z : zArray)
		{
			String display = "\t| [" + FormatUtil.FormatDouble (z, 1, 2, 1., false) + "] =>";

			display = display + " " + FormatUtil.FormatDouble (
				sphericalBesselFirstKindEstimator.smallJ (
					alpha,
					z
				), 1, 9, 1.
			) + " |";

			System.out.println (display + "|");
		}

		System.out.println ("\t|--------------------------------------------------------------------------||");

		System.out.println();
	}

	public static final void main (
		final String[] argumentArray)
		throws Exception
	{
		EnvManager.InitEnv ("");

		double[] zArray =
		{
			 0.00,
			 0.25,
			 0.50,
			 0.75,
			 1.00,
			 1.25,
			 1.50,
			 1.75,
			 2.00,
			 2.25,
			 2.50,
			 2.75,
			 3.00,
			 3.25,
			 3.50,
			 3.75,
			 4.00,
			 4.25,
			 4.50,
			 4.75,
			 5.00,
			 5.25,
			 5.50,
			 5.75,
			 6.00,
			 6.25,
			 6.50,
			 6.75,
			 7.00,
			 7.25,
			 7.50,
			 7.75,
			 8.00,
			 8.25,
			 8.50,
			 8.75,
			 9.00,
			 9.25,
			 9.50,
			 9.75,
		};

		Besselj (
			SphericalFirstEstimator.OrderMinusThree(),
			zArray
		);

		EnvManager.TerminateEnv();
	}
}
