{

	public static final void main (
		final String[] argumentArray)
		throws Exception
	{
		EnvManager.InitEnv ("");

		EulerIntegralSecondKind gammaFunction = new EulerIntegralSecondKind (null);

		System.out.println (gammaFunction.evaluate (5.));

		EnvManager.TerminateEnv();
	}
