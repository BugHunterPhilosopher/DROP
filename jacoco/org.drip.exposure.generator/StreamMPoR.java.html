public abstract class StreamMPoR implements org.drip.exposure.mpor.VariationMarginTradePaymentVertex
{
	private double _notional = java.lang.Double.NaN;
	private org.drip.product.rates.Stream _stream = null;

	protected StreamMPoR (
		final org.drip.product.rates.Stream stream,
		final double notional)
		throws java.lang.Exception
	{
		if (null == (_stream = stream) ||
			!org.drip.quant.common.NumberUtil.IsValid (_notional = notional))
		{
			throw new java.lang.Exception ("StreamMPoR Constructor => Invalid Inputs");
		}
	}

	/**
	 * Retrieve the Underlying Stream Instance
	 * 
	 * @return The Underlying Stream Instance
	 */

	public org.drip.product.rates.Stream stream()
	{
		return _stream;
	}

	/**
	 * Retrieve the Underlying Stream Notional
	 * 
	 * @return The Underlying Stream Notional
	 */

	public double notional()
	{
		return _notional;
	}
}
