public class ASeriesTerm extends org.drip.numerical.estimation.R1ToR1SeriesTerm
{

	/**
	 * Empty ASeriesTerm Constructor
	 */

	public ASeriesTerm()
	{
	}

	@Override public double value (
		final int order,
		final double z)
		throws java.lang.Exception
	{
		if (0 > order ||
			!org.drip.numerical.common.NumberUtil.IsValid (z))
		{
			throw new java.lang.Exception ("ASeriesTerm::value => Invalid Inputs");
		}

		if (0 == order)
		{
			return 0.5;
		}

		double numerator = 1.;
		double denominator = 1.;

		for (int index = 1; index <= order; ++index)
		{
			numerator = numerator * (z - index + 1);
			denominator = denominator * (z + index);
		}

		if (0. == denominator)
		{
			throw new java.lang.Exception ("ASeriesTerm::value => Invalid Inputs");
		}

		return numerator / denominator;
	}
}
