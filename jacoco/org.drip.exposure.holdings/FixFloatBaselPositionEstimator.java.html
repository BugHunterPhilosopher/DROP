public class FixFloatBaselPositionEstimator extends org.drip.exposure.holdings.PositionGroupEstimator
{
	private int _maturityDate = -1;
	private org.drip.state.identifier.OTCFixFloatLabel _otcFixFloatLabel = null;

	/**
	 * FixFloatBaselPositionEstimator Constructor
	 * 
	 * @param maturityDate The Fix Float Maturity Date
	 * @param otcFixFloatLabel OTC Fix Float Label
	 * 
	 * @throws java.lang.Exception Thrown if the Inputs are Invalid
	 */

	public FixFloatBaselPositionEstimator (
		final int maturityDate,
		final org.drip.state.identifier.OTCFixFloatLabel otcFixFloatLabel)
		throws java.lang.Exception
	{
		if (0 >= (_maturityDate = maturityDate) ||
			null == (_otcFixFloatLabel = otcFixFloatLabel))
		{
			throw new java.lang.Exception ("FixFloatBaselPositionEstimator Constructor => Invalid Inputs");
		}
	}

	/**
	 * Retrieve the Maturity Date
	 * 
	 * @return The Maturity Date
	 */

	public int maturityDate()
	{
		return _maturityDate;
	}

	/**
	 * Retrieve the OTC Fix Float Label
	 * 
	 * @return The OTC Fix Float Label
	 */

	public org.drip.state.identifier.OTCFixFloatLabel otcFixFloatLabel()
	{
		return _otcFixFloatLabel;
	}

	@Override public double variationMarginEstimate (
		final int vertexDate,
		final org.drip.exposure.universe.MarketPath marketPath)
		throws java.lang.Exception
	{
		if (null == marketPath)
		{
			throw new java.lang.Exception ("FixFloatBaselPositionEstimator::value => Invalid Inputs");
		}

		double dblTimeToHorizon = 1. * (_maturityDate - vertexDate) / 365.25;

		return dblTimeToHorizon > 0. ? dblTimeToHorizon *
			marketPath.marketVertex (vertexDate).latentStateValue (_otcFixFloatLabel) : 0.;
	}
}
