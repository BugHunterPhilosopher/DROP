public class IR1ClosesReconstitutor {

	public static final void main (
		final String[] args)
		throws Exception
	{
		EnvManager.InitEnv ("");

		int iSpotDateIndex = 0;
		int iLastPriceIndex = 2;
		int iFuturesExpiryDateIndex = 8;
		String strFeedLocation = "C:\\DROP\\Daemons\\Feeds\\FundingFutures\\IR1_ClosesFormatted.csv";

		FundingFuturesClosesReconstitutor.RegularizeCloses (
			strFeedLocation,
			iSpotDateIndex,
			iLastPriceIndex,
			iFuturesExpiryDateIndex
		);

		EnvManager.TerminateEnv();
	}
}
