public class ILP
{
	private org.drip.linearprogram.canonical.ILPObjective _objective = null;
	private org.drip.linearprogram.canonical.ILPConstraint _constraint = null;

	/**
	 * ILP Constructor
	 * 
	 * @param objective The Objective Term
	 * @param constraint The Constraint Term
	 * 
	 * @throws java.lang.Exception Thrown if the Inputs are Invalid
	 */

	public ILP (
		final org.drip.linearprogram.canonical.ILPObjective objective,
		final org.drip.linearprogram.canonical.ILPConstraint constraint)
		throws java.lang.Exception
	{
		if (null == (_objective = objective) ||
			null == (_constraint = constraint) ||
			_objective.dimension() != _constraint.dimension()
		)
		{
			throw new java.lang.Exception (
				"ILP Constructor => Invalid Inputs"
			);
		}
	}

	/**
	 * Retrieve the Objective Term
	 * 
	 * @return The Objective Term
	 */

	public org.drip.linearprogram.canonical.ILPObjective objective()
	{
		return _objective;
	}

	/**
	 * Retrieve the Constraint Term
	 * 
	 * @return The Constraint Term
	 */

	public org.drip.linearprogram.canonical.ILPConstraint constraint()
	{
		return _constraint;
	}
}
